 function matrix4 normColumns(matrix4 m){
     double[4;4] val = m.getMatrix();
     loop(int i = 0; i < 4; i = i + 1){
         double sum = 0;
         loop(int j = 0; j < 4; j = j + 1){
             sum = sum + val[i,j];
         }
         loop(int j = 0; j < 4; j = j + 1){
             val[i;j] = val[i,j] / sum;
         }
     }
     return Matrix4(val);
 }
